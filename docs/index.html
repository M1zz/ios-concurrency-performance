Before / After

// ❌ Before — 퍼플 경고, 크래시 가능
URLSession.shared.dataTask(with: url) { data, _, _ in
    self.imageView.image = UIImage(data: data!) // 백그라운드 스레드!
}.resume()

// ✅ After — async/await + @MainActor
func loadImage() async throws {
    let (data, _) = try await URLSession.shared.data(from: url)
    await MainActor.run {
        self.imageView.image = UIImage(data: data)
    }
}
// ❌ Before — 48MB
let image = UIImage(data: try! Data(contentsOf: url))

// ✅ After — 0.04MB (1200배 절약)
let image = downsample(imageAt: url, to: CGSize(width: 100, height: 100))

이 단계를 마치면

  • async/await로 비동기 코드를 동기처럼 작성할 수 있다
  • Actor로 Data Race를 컴파일러 수준에서 방지한다
  • ImageIO로 이미지 메모리를 1200배 줄일 수 있다
  • Combine으로 복잡한 이벤트 흐름을 선언적으로 처리한다
